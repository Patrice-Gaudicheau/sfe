import * as THREE from "https://unpkg.com/three@0.165.0/build/three.module.js";

const sceneContainer = document.getElementById("sceneContainer");
const playPauseButton = document.getElementById("playPauseButton");
const timeSpeedInput = document.getElementById("timeSpeed");
const timeSpeedValue = document.getElementById("timeSpeedValue");
const datePreset = document.getElementById("datePreset");
const seasonButtons = document.querySelectorAll("[data-season]");
const bodyName = document.getElementById("bodyName");
const bodyType = document.getElementById("bodyType");
const bodyDistance = document.getElementById("bodyDistance");
const bodyOrbitPeriod = document.getElementById("bodyOrbitPeriod");
const bodyRotationPeriod = document.getElementById("bodyRotationPeriod");
const bodyDescription = document.getElementById("bodyDescription");

const state = {
  isPlaying: true,
  timeSpeed: Number(timeSpeedInput.value),
  selectedSeason: "live",
};

const renderer = new THREE.WebGLRenderer({
  antialias: true,
  alpha: false,
});
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(sceneContainer.clientWidth, sceneContainer.clientHeight);
renderer.outputColorSpace = THREE.SRGBColorSpace;
sceneContainer.appendChild(renderer.domElement);

const scene = new THREE.Scene();
scene.background = new THREE.Color(0x02050d);

const camera = new THREE.PerspectiveCamera(
  52,
  sceneContainer.clientWidth / sceneContainer.clientHeight,
  0.1,
  2000
);
camera.position.set(0, 28, 78);

const ambientLight = new THREE.AmbientLight(0x8aa4ff, 0.6);
scene.add(ambientLight);

const sunLight = new THREE.PointLight(0xffe1a8, 2.8, 0, 2);
sunLight.position.set(0, 0, 0);
scene.add(sunLight);

const rimLight = new THREE.DirectionalLight(0x7aa2ff, 0.8);
rimLight.position.set(-40, 20, 25);
scene.add(rimLight);

const solarGroup = new THREE.Group();
scene.add(solarGroup);

const orbitGuideMaterial = new THREE.MeshBasicMaterial({
  color: 0x2f4f88,
  transparent: true,
  opacity: 0.22,
  wireframe: true,
});

const orbitGuideGeometry = new THREE.TorusGeometry(18, 0.05, 6, 120);
const orbitGuide = new THREE.Mesh(orbitGuideGeometry, orbitGuideMaterial);
orbitGuide.rotation.x = Math.PI / 2;
solarGroup.add(orbitGuide);

const sunMaterial = new THREE.MeshStandardMaterial({
  map: createSunTexture(),
  emissive: new THREE.Color(0xffa83a),
  emissiveIntensity: 1.15,
  roughness: 0.8,
  metalness: 0,
});

const sunMesh = new THREE.Mesh(
  new THREE.SphereGeometry(5.8, 48, 48),
  sunMaterial
);
solarGroup.add(sunMesh);

const accentPlanet = new THREE.Mesh(
  new THREE.SphereGeometry(1.6, 32, 32),
  new THREE.MeshStandardMaterial({
    map: createPlanetTexture({
      base: "#4d86ff",
      accent: "#7ecf7a",
      cloud: "#ffffff",
    }),
    roughness: 1,
    metalness: 0,
  })
);
accentPlanet.position.set(18, 0, 0);
solarGroup.add(accentPlanet);

const secondaryPlanet = new THREE.Mesh(
  new THREE.SphereGeometry(2.4, 32, 32),
  new THREE.MeshStandardMaterial({
    map: createPlanetTexture({
      base: "#c58e4b",
      accent: "#9d5c32",

    }),
    roughness: 1,
    metalness: 0,
  })
);
secondaryPlanet.position.set(-27, 0, -10);
solarGroup.add(secondaryPlanet);

const starfield = createStarfield(1800, 900);

const clock = new THREE.Clock();

function animate() {
  requestAnimationFrame(animate);

  const delta = clock.getDelta();

  if (state.isPlaying) {
    const speedFactor = 0.12 + state.timeSpeed * 0.012;
    sunMesh.rotation.y += delta * 0.2;
    accentPlanet.position.x = Math.cos(clock.elapsedTime * 0.25 * speedFactor) * 18;
    accentPlanet.position.z = Math.sin(clock.elapsedTime * 0.25 * speedFactor) * 18;
    accentPlanet.rotation.y += delta * 0.9;

    secondaryPlanet.position.x =
      Math.cos(clock.elapsedTime * 0.12 * speedFactor + 1.4) * 29;
    secondaryPlanet.position.z =
      Math.sin(clock.elapsedTime * 0.12 * speedFactor + 1.4) * 29;
    secondaryPlanet.rotation.y += delta * 0.6;
  }

  renderer.render(scene, camera);
}

function createSunTexture() {
  const size = 256;
  const canvas = document.createElement("canvas");
  canvas.width = size;
  canvas.height = size;
  const ctx = canvas.getContext("2d");

  const gradient = ctx.createRadialGradient(
    size * 0.5,
    size * 0.5,
    size * 0.08,
    size * 0.5,
    size * 0.5,
    size * 0.5
  );
  gradient.addColorStop(0, "#fff8c9");
  gradient.addColorStop(0.3, "#ffc85a");
  gradient.addColorStop(0.68, "#ff8c2e");
  gradient.addColorStop(1, "#7f2600");
  ctx.fillStyle = gradient;
  ctx.fillRect(0, 0, size, size);

  for (let i = 0; i < 550; i += 1) {
    const x = Math.random() * size;
    const y = Math.random() * size;
    const radius = Math.random() * 18 + 4;
    ctx.fillStyle = `rgba(255, ${160 + Math.random() * 70}, 0, ${0.06 + Math.random() * 0.08})`;
    ctx.beginPath();
    ctx.arc(x, y, radius, 0, Math.PI * 2);
    ctx.fill();
  }

  const texture = new THREE.CanvasTexture(canvas);
  texture.colorSpace = THREE.SRGBColorSpace;
  return texture;
}

function createPlanetTexture({ base, accent, cloud }) {
  const size = 256;
  const canvas = document.createElement("canvas");
  canvas.width = size;
  canvas.height = size;
  const ctx = canvas.getContext("2d");

  ctx.fillStyle = base;
  ctx.fillRect(0, 0, size, size);

  for (let i = 0; i < 220; i += 1) {
    ctx.fillStyle = `${accent}${Math.floor(100 + Math.random() * 100).toString(16).padStart(2, "0")}`;
    ctx.beginPath();
    ctx.arc(
      Math.random() * size,
      Math.random() * size,
      Math.random() * 24 + 6,
      0,
      Math.PI * 2
    );
    ctx.fill();
  }

  for (let i = 0; i < 120; i += 1) {
    ctx.fillStyle = `${cloud}${Math.floor(60 + Math.random() * 80).toString(16).padStart(2, "0")}`;
    ctx.beginPath();
    ctx.ellipse(
      Math.random() * size,
      Math.random() * size,
      Math.random() * 18 + 8,
      Math.random() * 8 + 4,
      Math.random() * Math.PI,
      0,
      Math.PI * 2
    );
    ctx.fill();
  }

  const texture = new THREE.CanvasTexture(canvas);
  texture.colorSpace = THREE.SRGBColorSpace;
  return texture;
}

function createStarfield(count, spread) {
  const positions = new Float32Array(count * 3);

  for (let i = 0; i < count; i += 1) {
    const radius = spread * (0.55 + Math.random() * 0.45);
    const theta = Math.random() * Math.PI * 2;
    const phi = Math.acos(2 * Math.random() - 1);
    const index = i * 3;

    positions[index] = radius * Math.sin(phi) * Math.cos(theta);
    positions[index + 1] = radius * Math.cos(phi);
    positions[index + 2] = radius * Math.sin(phi) * Math.sin(theta);
  }

  const geometry = new THREE.BufferGeometry();
  geometry.setAttribute("position", new THREE.BufferAttribute(positions, 3));

  const material = new THREE.PointsMaterial({
    color: 0xffffff,
    size: 1.6,
    sizeAttenuation: true,
    transparent: true,
    opacity: 0.88,
  });

  return new THREE.Points(geometry, material);
}

# SolarSystem3D

SolarSystem3D is a static Three.js browser app scaffold for an educational solar system simulator. This first task establishes the simulator-first layout, a full-screen 3D scene shell, responsive control areas, a selected-body information panel placeholder, baseline lighting, resize handling, and a procedurally generated star background.

## Current status

This is the initial scene shell, not the fully completed simulator yet. The current implementation includes:

- A simulator-first page with the 3D scene visible on first load
- Responsive top control regions for simulation and view settings
- Placeholder controls for:
  - play/pause
  - time speed
  - date and season presets
  - camera presets
  - label visibility
  - orbit visibility
  - scale mode
- A selected-body panel scaffold
- A baseline Three.js scene with:
  - perspective camera
  - WebGL renderer
  - ambient, point, and directional lighting
  - resize handling
  - procedural starfield
  - temporary solar-system placeholder meshes

Later tasks will add the full Sun, all eight planets, Earth's Moon, visible orbit paths for each body, procedural body textures, labels, selection, camera focus behavior, seasons, and additional accessibility polish.

## Run instructions

This app uses a pinned browser ES module import for Three.js:

- `https://unpkg.com/three@0.165.0/build/three.module.js`

Because some browsers restrict ES module loading from `file://`, the most reliable way to run the app is with a simple static server.

### Recommended

## Accessibility notes

- Controls use semantic HTML elements such as `button`, `input`, `select`, `label`, `section`, and `aside`.
- Visible focus styles are included for keyboard navigation.
- The 3D scene remains the primary surface instead of being hidden behind a landing page.
- Controls are visible without hover-only interaction.
- Placeholder controls are keyboard reachable even before all behaviors are fully wired.

## Astronomy simplifications

This project is intended as an educational approximation rather than a precision astronomy tool.

Current and planned simplifications include:

- Simplified orbital layout for readability
- Educational season positioning rather than ephemeris-grade dates
- A clear note that seasons are caused by Earth's axial tilt relative to sunlight, not by Earth being closer to the Sun
- Procedural textures instead of downloaded planetary imagery
