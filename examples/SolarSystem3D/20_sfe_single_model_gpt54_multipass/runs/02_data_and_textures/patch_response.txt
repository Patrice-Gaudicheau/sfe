const clock = new THREE.Clock();

function animate() {
  requestAnimationFrame(animate);

  const delta = clock.getDelta();

  if (state.isPlaying) {
    const speedFactor = 0.12 + state.timeSpeed * 0.012;

    bodyMeshes.forEach((mesh, index) => {
      const body = mesh.userData.body;
      mesh.rotation.y += delta * (0.15 + index * 0.03);

      if (body.name !== "Sun" && !body.parent) {
        const angle =
          clock.elapsedTime * speedFactor * (0.06 + index * 0.008) + index;
        mesh.position.x = Math.cos(angle) * body.orbitRadius;
        mesh.position.z = Math.sin(angle) * body.orbitRadius;
      }

      if (body.parent === "Earth") {
        const earthMesh = bodyMeshes.find(
          (candidate) => candidate.userData.body.name === "Earth"
        );
        if (earthMesh) {
          const angle = clock.elapsedTime * speedFactor * 0.9;
          mesh.position.x =
            earthMesh.position.x + Math.cos(angle) * body.orbitRadius;
          mesh.position.z =
            earthMesh.position.z + Math.sin(angle) * body.orbitRadius;
        }
      }
    });
  }

  renderer.render(scene, camera);
}

function createBodyPreviewMeshes(bodies) {
  const meshGroup = [];

  bodies.forEach((body, index) => {
    const geometry = new THREE.SphereGeometry(
      body.visualRadius,
      body.name === "Sun" ? 48 : 32,
      body.name === "Sun" ? 48 : 32
    );

    const materialOptions = {
      map: getBodyTexture(body),
      roughness: body.name === "Sun" ? 0.82 : 1,
      metalness: 0,
    };

    if (body.name === "Sun") {
      materialOptions.emissive = new THREE.Color(0xffa83a);
      materialOptions.emissiveIntensity = 1.15;
    }

    const mesh = new THREE.Mesh(
      geometry,
      new THREE.MeshStandardMaterial(materialOptions)
    );

    mesh.userData.body = body;

    if (body.name === "Sun") {
      mesh.position.set(0, 0, 0);
    } else if (!body.parent) {
      mesh.position.set(body.orbitRadius, 0, 0);
    } else {
      const earth = bodies.find((candidate) => candidate.name === "Earth");
      mesh.position.set((earth?.orbitRadius || 18) + body.orbitRadius, 0, 0);
    }

    mesh.userData.previewIndex = index;
    meshGroup.push(mesh);
  });

  return meshGroup;
}

function getBodyTexture(body) {
  const cacheKey = `${body.name}-${JSON.stringify(body.textureStyle)}`;

  if (textureCache.has(cacheKey)) {
    return textureCache.get(cacheKey);
  }

  let texture;

  switch (body.textureStyle.preset) {
    case "sun":
      texture = createSunTexture(body.textureStyle);
      break;
    case "earth":
      texture = createEarthTexture(body.textureStyle);
      break;
    case "moon":
      texture = createMoonTexture(body.textureStyle);
      break;
    case "mars":
      texture = createMarsTexture(body.textureStyle);
      break;
    case "jupiter":
      texture = createGasGiantTexture(body.textureStyle, {
        addStorm: true,
      });
      break;
    case "saturn":
      texture = createGasGiantTexture(body.textureStyle, {
        addStorm: false,
      });
      break;
    case "ice":
    case "neptune":
      texture = createIceGiantTexture(body.textureStyle);
      break;
    case "rocky":
    default:
      texture = createRockyPlanetTexture(body.textureStyle);
      break;
  }

  textureCache.set(cacheKey, texture);
  return texture;
}

function finalizeTexture(canvas) {
  const texture = new THREE.CanvasTexture(canvas);
  texture.colorSpace = THREE.SRGBColorSpace;
  texture.wrapS = THREE.RepeatWrapping;
  texture.wrapT = THREE.ClampToEdgeWrapping;
  texture.needsUpdate = true;
  return texture;
}

function randomBetween(min, max) {
  return min + Math.random() * (max - min);
}

function drawNoiseSpeckles(
  ctx,
  size,
  colors,
  count,
  minRadius,
  maxRadius,
  opacity
) {
  for (let i = 0; i < count; i += 1) {
    ctx.fillStyle = hexToRgba(
      colors[Math.floor(Math.random() * colors.length)],
      randomBetween(opacity * 0.55, opacity)
    );
    ctx.beginPath();
    ctx.arc(
      Math.random() * size,
      Math.random() * size,
      randomBetween(minRadius, maxRadius),
      0,
      Math.PI * 2
    );
    ctx.fill();
  }
}

function drawBandLayer(
  ctx,
  size,
  colors,
  bandCount,
  opacity,
  blur = 0,
  waveStrength = 0
) {
  const bandHeight = size / bandCount;

  ctx.save();
  ctx.globalAlpha = opacity;
  ctx.filter = blur > 0 ? `blur(${blur}px)` : "none";

  for (let i = 0; i < bandCount; i += 1) {
    const y = i * bandHeight;
    const height = bandHeight * randomBetween(0.8, 1.35);
    const gradient = ctx.createLinearGradient(0, y, 0, y + height);
    const colorA = colors[i % colors.length];
    const colorB = colors[(i + 1) % colors.length];
    gradient.addColorStop(0, colorA);
    gradient.addColorStop(1, colorB);
    ctx.fillStyle = gradient;
    ctx.fillRect(0, y, size, height);

    if (waveStrength > 0) {
      ctx.strokeStyle = hexToRgba(colors[i % colors.length], 0.3);
      ctx.lineWidth = randomBetween(3, 8);
      ctx.beginPath();
      for (let x = 0; x <= size; x += 12) {
        const waveY =
          y +
          height * 0.5 +
          Math.sin((x / size) * Math.PI * 2 + i * 0.7) *
            randomBetween(2, waveStrength);
        if (x === 0) {
          ctx.moveTo(x, waveY);
        } else {
          ctx.lineTo(x, waveY);
        }
      }
      ctx.stroke();

function hexToRgba(hex, alpha) {
  const normalized = hex.replace("#", "");
  const value =
    normalized.length === 3
      ? normalized
          .split("")
          .map((character) => character + character)
          .join("")
      : normalized;

  const r = Number.parseInt(value.slice(0, 2), 16);
  const g = Number.parseInt(value.slice(2, 4), 16);
  const b = Number.parseInt(value.slice(4, 6), 16);

  return `rgba(${r}, ${g}, ${b}, ${alpha})`;
}

function lightenColor(hex, amount) {
  return adjustColor(hex, amount);
}

function darkenColor(hex, amount) {
  return adjustColor(hex, -amount);
}

function adjustColor(hex, amount) {
  const normalized = hex.replace("#", "");
  const parts =
    normalized.length === 3
      ? normalized
          .split("")
          .map((character) => character + character)
      : [
          normalized.slice(0, 2),
          normalized.slice(2, 4),
          normalized.slice(4, 6),
        ];

  const adjusted = parts.map((part) => {
    const base = Number.parseInt(part, 16);
    const next = Math.max(0, Math.min(255, Math.round(base + 255 * amount)));
    return next.toString(16).padStart(2, "0");
  });

  return `#${adjusted.join("")}`;
}

function createStarfield(count, spread) {
  const positions = new Float32Array(count * 3);

  for (let i = 0; i < count; i += 1) {
    const radius = spread * (0.55 + Math.random() * 0.45);
    const theta = Math.random() * Math.PI * 2;
    const phi = Math.acos(2 * Math.random() - 1);
    const index = i * 3;

    positions[index] = radius * Math.sin(phi) * Math.cos(theta);
    positions[index + 1] = radius * Math.cos(phi);
    positions[index + 2] = radius * Math.sin(phi) * Math.sin(theta);
  }

  const geometry = new THREE.BufferGeometry();
  geometry.setAttribute("position", new THREE.BufferAttribute(positions, 3));

  const material = new THREE.PointsMaterial({
    color: 0xffffff,
    size: 1.6,
    sizeAttenuation: true,
    transparent: true,
    opacity: 0.88,
  });

  return new THREE.Points(geometry, material);
}

- body textures are selected by per-body `textureStyle.preset`
- the selected-body panel starts with real Sun metadata
- season preset changes preserve the currently selected body panel content while appending a simplified preset note

This is still a task-2 integration pass, not the full body-orbit-label-selection implementation planned for later tasks.

## Reuse and performance notes

The texture pipeline is designed for reuse in later rendering tasks:

- textures are generated once per body style and cached in memory
- textures are not recreated during animation frames
- helper functions are generic enough to support future planet mesh creation and ring/body extensions
- the preview scene continues to use one main animation loop

## Accessibility notes

- Controls use semantic HTML elements such as `button`, `input`, `select`, `label`, `section`, and `aside`.
- Visible focus styles are included for keyboard navigation.
- The scene remains the primary first-screen surface.
- Controls are visible without hover-only interaction.
- Core controls shown in the UI remain keyboard reachable.
- The selected-body panel uses readable labels for educational metadata fields.

## Astronomy simplifications

This project is an educational approximation rather than a precision astronomy tool.

Current and planned simplifications include:

- Rounded approximate orbital and rotation values
- Simplified orbit radii for readable scene layout
- Educational visual radii rather than physically exact render scaling
- Procedural surfaces that suggest major planetary features instead of photoreal imagery
- Season presets that will use simplified positions rather than ephemeris-grade calculations
- A clear explanation that seasons are caused by Earth's axial tilt relative to sunlight, not by Earth being closer to or farther from the Sun
