+(function () {
+  const noteTypeField = document.querySelector("#note-type");
+  const checklistBuilder = document.querySelector("#checklist-builder");
+  const checklistItems = document.querySelector("#checklist-items");
+  const addChecklistItemButton = document.querySelector("#add-checklist-item");
+  const viewButtons = document.querySelectorAll(".view-button");
+  const notesStatus = document.querySelector("#notes-status");
+  const noteForm = document.querySelector("#note-form");
+
+  function updateChecklistVisibility() {
+    const isChecklist = noteTypeField.value === "checklist";
+    checklistBuilder.hidden = !isChecklist;
+  }
+
+  function createChecklistRow() {
+    const row = document.createElement("div");
+    row.className = "checklist-item-row";
+
+    const checkbox = document.createElement("input");
+    checkbox.type = "checkbox";
+    checkbox.setAttribute("aria-label", "Checklist item complete state");
+
+    const textInput = document.createElement("input");
+    textInput.type = "text";
+    textInput.placeholder = "Checklist item text";
+    textInput.setAttribute("aria-label", "Checklist item text");
+
+    const removeButton = document.createElement("button");
+    removeButton.type = "button";
+    removeButton.className = "secondary-button";
+    removeButton.textContent = "Remove";
+    removeButton.addEventListener("click", function () {
+      row.remove();
+    });
+
+    row.appendChild(checkbox);
+    row.appendChild(textInput);
+    row.appendChild(removeButton);
+
+    return row;
+  }
+
+  function setActiveView(viewName) {
+    viewButtons.forEach(function (button) {
+      const isActive = button.dataset.view === viewName;
+      button.classList.toggle("is-active", isActive);
+      button.setAttribute("aria-pressed", String(isActive));
+    });
+
+    notesStatus.textContent =
+      viewName === "archived"
+        ? "Archive view selected. Archived notes will render here in a later task."
+        : "Active notes view selected. Your saved notes will render here in a later task.";
+  }
+
+  if (noteTypeField && checklistBuilder) {
+    noteTypeField.addEventListener("change", updateChecklistVisibility);
+    updateChecklistVisibility();
+  }
+
+  if (addChecklistItemButton && checklistItems) {
+    addChecklistItemButton.addEventListener("click", function () {
+      checklistItems.appendChild(createChecklistRow());
+    });
+  }
+
+  viewButtons.forEach(function (button) {
+    button.addEventListener("click", function () {
+      setActiveView(button.dataset.view || "active");
+    });
+  });
+
+  if (noteForm) {
+    noteForm.addEventListener("submit", function (event) {
+      event.preventDefault();
+      notesStatus.textContent =
+        "Scaffold only: note saving will be implemented in the next task.";
+    });
+
+    noteForm.addEventListener("reset", function () {
+      window.setTimeout(function () {
+        while (checklistItems.children.length > 1) {
+          checklistItems.removeChild(checklistItems.lastElementChild);
+        }
+
+        const firstRow = checklistItems.firstElementChild;
+        if (firstRow) {
+          const checkbox = firstRow.querySelector('input[type="checkbox"]');
+          const textInput = firstRow.querySelector('input[type="text"]');
+
+          if (checkbox) {
+            checkbox.checked = false;
+          }
+
+          if (textInput) {
+            textInput.value = "";
+          }
+        }
+
+        updateChecklistVisibility();
+        notesStatus.textContent =
+          "Form reset. Create a text note or checklist note to populate this area in the next task.";
+      }, 0);
+    });
+  }
+
+  if (checklistItems) {
+    const firstTextInput = checklistItems.querySelector('input[type="text"]');
+    const firstCheckbox = checklistItems.querySelector('input[type="checkbox"]');
+
+    if (firstCheckbox) {
+      firstCheckbox.disabled = false;
+    }
+
+    if (firstTextInput) {
+      firstTextInput.value = "";
+    }
+  }
+
+  setActiveView("active");
+})();

+# NoteKeeper
+
+NoteKeeper is a small static browser app for capturing notes and checklist ideas without any server or build step.
+
+## Run the app
+
+1. Open the `app/` folder.
+2. Double-click `index.html`, or open it directly in a modern browser.
+3. No installation, package manager, or local server is required.
+
+## Current scaffold contents
+
+This initial scaffold includes:
+
+- a NoteKeeper app shell that opens directly to the main workspace
+- a search field in the header
+- active and archived note view controls
+- a note creation form with fields for title, body, labels, and note type
+- a checklist item editing area with add and remove controls
+- a placeholder notes area for future rendered note cards
+- baseline responsive styling for phone, tablet, and desktop widths
+
+## Current behavior
+
+- switching between Active notes and Archived notes updates the visible scaffold status
+- switching the note type shows or hides the checklist editor area
+- adding checklist rows works in the form scaffold
+- submitting the form is intentionally a placeholder until note persistence and CRUD are implemented in later tasks
+
+## Manual review steps
+
+1. Open `index.html` and confirm the page title and visible app name are NoteKeeper.
+2. Confirm the header includes a search field plus Active notes and Archived notes buttons.
+3. Confirm the form includes accessible labels for title, body, labels, and note type.
+4. Switch note type to Checklist note and verify the checklist item area is visible.
+5. Use the Add checklist item button and confirm another checklist row appears.
+6. Confirm the notes area shows an empty placeholder card ready for later rendering.
