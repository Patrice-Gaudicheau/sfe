+const STORAGE_KEY = "notekeeper.notes.v1";
+
+const state = {
+  notes: loadNotes(),
+  view: "active"
+};
+
+const elements = {
+  form: document.getElementById("note-form"),
+  noteId: document.getElementById("note-id"),
+  title: document.getElementById("note-title"),
+  body: document.getElementById("note-body"),
+  cancelEdit: document.getElementById("cancel-edit"),
+  notesList: document.getElementById("notes-list"),
+  emptyState: document.getElementById("empty-state"),
+  statusText: document.getElementById("status-text"),
+  viewButtons: document.querySelectorAll(".view-button"),
+  cardTemplate: document.getElementById("note-card-template")
+};
+
+function loadNotes() {
+  try {
+    const raw = window.localStorage.getItem(STORAGE_KEY);
+
+    if (!raw) {
+      return [];
+    }
+
+    const parsed = JSON.parse(raw);
+
+    if (!Array.isArray(parsed)) {
+      return [];
+    }
+
+    return parsed
+      .filter((note) => note && typeof note === "object")
+      .map((note) => ({
+        id: typeof note.id === "string" ? note.id : crypto.randomUUID(),
+        title: typeof note.title === "string" ? note.title : "",
+        body: typeof note.body === "string" ? note.body : "",
+        type: "text",
+        archived: Boolean(note.archived),
+        createdAt: typeof note.createdAt === "string" ? note.createdAt : new Date().toISOString(),
+        updatedAt: typeof note.updatedAt === "string" ? note.updatedAt : new Date().toISOString()
+      }));
+  } catch (error) {
+    return [];
+  }
+}
+
+function saveNotes() {
+  window.localStorage.setItem(STORAGE_KEY, JSON.stringify(state.notes));
+}
+
+function resetForm() {
+  elements.form.reset();
+  elements.noteId.value = "";
+  elements.cancelEdit.hidden = true;
+}
+
+function getFilteredNotes() {
+  return state.notes
+    .filter((note) => state.view === "archived" ? note.archived : !note.archived)
+    .sort((a, b) => new Date(b.updatedAt) - new Date(a.updatedAt));
+}
+
+function formatDate(value) {
+  const date = new Date(value);
+
+  if (Number.isNaN(date.getTime())) {
+    return "Saved recently";
+  }
+
+  return date.toLocaleString();
+}
+
+function renderNotes() {
+  const visibleNotes = getFilteredNotes();
+  elements.notesList.innerHTML = "";
+
+  const isArchivedView = state.view === "archived";
+  const emptyTitle = isArchivedView ? "No archived notes" : "No notes yet";
+  const emptyBody = isArchivedView
+    ? "Archived notes will appear here when you move them out of the main view."
+    : "Create your first note to start building a local notebook.";
+
+  elements.emptyState.querySelector("h3").textContent = emptyTitle;
+  elements.emptyState.querySelector("p").textContent = emptyBody;
+
+  if (visibleNotes.length === 0) {
+    elements.emptyState.hidden = false;
+    elements.statusText.textContent = isArchivedView
+      ? "You are viewing archived notes."
+      : "Your saved notes appear here.";
+    return;
+  }
+
+  elements.emptyState.hidden = true;
+  elements.statusText.textContent = `${visibleNotes.length} note${visibleNotes.length === 1 ? "" : "s"} shown`;
+
+  visibleNotes.forEach((note) => {
+    const fragment = elements.cardTemplate.content.cloneNode(true);
+    const card = fragment.querySelector(".note-card");
+
+    const title = document.createElement("h3");
+    title.textContent = note.title.trim() || "Untitled note";
+
+    const body = document.createElement("p");
+    body.textContent = note.body.trim() || "This note has no body text.";
+    if (!note.body.trim()) {
+      body.classList.add("visually-muted");
+    }
+
+    const meta = document.createElement("p");
+    meta.className = "note-meta";
+    meta.textContent = `Updated ${formatDate(note.updatedAt)}`;
+
+    const actions = document.createElement("div");
+    actions.className = "note-actions";
+
+    const editButton = document.createElement("button");
+    editButton.type = "button";
+    editButton.className = "card-button";
+    editButton.textContent = "Edit";
+    editButton.addEventListener("click", () => startEdit(note.id));
+
+    const archiveButton = document.createElement("button");
+    archiveButton.type = "button";
+    archiveButton.className = "card-button";
+    archiveButton.textContent = note.archived ? "Restore" : "Archive";
+    archiveButton.addEventListener("click", () => toggleArchive(note.id));
+
+    const deleteButton = document.createElement("button");
+    deleteButton.type = "button";
+    deleteButton.className = "card-button delete-button";
+    deleteButton.textContent = "Delete";
+    deleteButton.addEventListener("click", () => deleteNote(note.id));
+
+    actions.append(editButton, archiveButton, deleteButton);
+    card.append(title, body, meta, actions);
+    elements.notesList.appendChild(fragment);
+  });
+}
+
+function createNote({ title, body }) {
+  const now = new Date().toISOString();
+
+  state.notes.unshift({
+    id: crypto.randomUUID(),
+    title,
+    body,
+    type: "text",
+    archived: false,
+    createdAt: now,
+    updatedAt: now
+  });
+
+  saveNotes();
+  renderNotes();
+}
+
+function updateNote(id, updates) {
+  const note = state.notes.find((item) => item.id === id);
+
+  if (!note) {
+    return;
+  }
+
+  note.title = updates.title;
+  note.body = updates.body;
+  note.updatedAt = new Date().toISOString();
+
+  saveNotes();
+  renderNotes();
+}
+
+function startEdit(id) {
+  const note = state.notes.find((item) => item.id === id);
+
+  if (!note) {
+    return;
+  }
+
+  elements.noteId.value = note.id;
+  elements.title.value = note.title;
+  elements.body.value = note.body;
+  elements.cancelEdit.hidden = false;
+  elements.title.focus();
+}
+
+function deleteNote(id) {
+  state.notes = state.notes.filter((note) => note.id !== id);
+  saveNotes();
+  renderNotes();
+
+  if (elements.noteId.value === id) {
+    resetForm();
+  }
+}
+
+function toggleArchive(id) {
+  const note = state.notes.find((item) => item.id === id);
+
+  if (!note) {
+    return;
+  }
+
+  note.archived = !note.archived;
+  note.updatedAt = new Date().toISOString();
+  saveNotes();
+  renderNotes();
+
+  if (elements.noteId.value === id) {
+    resetForm();
+  }
+}
+
+elements.form.addEventListener("submit", (event) => {
+  event.preventDefault();
+
+  const payload = {
+    title: elements.title.value.trim(),
+    body: elements.body.value.trim()
+  };
+
+  const editingId = elements.noteId.value;
+
+  if (editingId) {
+    updateNote(editingId, payload);
+  } else {
+    createNote(payload);
+  }
+
+  resetForm();
+});
+
+elements.cancelEdit.addEventListener("click", () => {
+  resetForm();
+});
+
+elements.viewButtons.forEach((button) => {
+  button.addEventListener("click", () => {
+    state.view = button.dataset.view;
+
+    elements.viewButtons.forEach((item) => {
+      const isActive = item === button;
+      item.classList.toggle("is-active", isActive);
+      item.setAttribute("aria-pressed", String(isActive));
+    });
+
+    renderNotes();
+  });
+});
+
+renderNotes();

+# NoteKeeper
+
+NoteKeeper is a small browser-only notes app that runs directly from local files. No server, build step, or dependencies are required.
+
+## Run
+
+1. Open the `app/` folder.
+2. Double-click `index.html`, or open it directly in a modern browser.
+
+## Current features
+
+- Create plain text notes with an optional title and body
+- Edit existing notes
+- Delete notes
+- Archive and restore notes
+- Switch between active and archived note views
+- Persist notes in `localStorage`
+- Recover safely when stored note data is missing or malformed
+
+## Storage
+
+Notes are saved in the browser's `localStorage` under a NoteKeeper-specific key: `notekeeper.notes.v1`.
+
+## Manual verification
+
+1. Open `index.html`.
+2. Create a note with a title and body, then save it.
+3. Edit the note and save again.
+4. Delete a note and confirm it disappears.
+5. Create another note, reload the browser tab, and confirm the note still appears.
+6. Archive a note, switch to the archived view, and restore it back to active notes.
