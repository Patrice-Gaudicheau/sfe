   function startEditing(noteId) {

   }
 
   function upsertNote(noteData) {
     const timestamp = new Date().toISOString();
     const isChecklist = noteData.type === "checklist";
     const savedNoteType = isChecklist ? "text" : noteData.type;
     const savedBody = isChecklist ? noteData.body : noteData.body;
 
     if (state.editingNoteId) {
       state.notes = state.notes.map(function (note) {
         if (note.id !== state.editingNoteId) {
           return note;
         }
 
         return {
           id: note.id,
           title: noteData.title,
           body: savedBody || noteData.body,
           type: savedNoteType,
           labels: noteData.labels,
           checklistItems: note.checklistItems,
           archived: note.archived,
           pinned: note.pinned,
           createdAt: note.createdAt,
           updatedAt: timestamp
         };
       });
 
       notesStatus.textContent = isChecklist
         ? "Checklist saving is not available yet. Your note was saved as a plain text note for now."
         : "Note updated and saved to this browser.";
     } else {
       state.notes.unshift({
         id: createNoteId(),
         title: noteData.title,
         body: savedBody || noteData.body,
         type: savedNoteType,
         labels: noteData.labels,
         checklistItems: [],
         archived: false,
         pinned: false,
         createdAt: timestamp,
         updatedAt: timestamp
       });
 
       notesStatus.textContent = isChecklist
         ? "Checklist saving is not available yet. Your note was saved as a plain text note for now."
         : "Note saved to this browser.";
     }
   }
 
   function deleteNote(noteId) {
     state.notes = state.notes.filter(function (note) {
       return note.id !== noteId;
     });
 
     if (state.editingNoteId === noteId) {
       resetForm();
     }
 
     if (!saveNotes()) {
       notesStatus.textContent = "Note deleted in memory, but local storage was not available.";
       renderNotes();
       return;
     }
 
     renderNotes();
     notesStatus.textContent = "Note deleted.";
   }
 
   function setActiveView(viewName) {
     state.activeView = viewName;
 
     viewButtons.forEach(function (button) {
       const isActive = button.dataset.view === viewName;
       button.classList.toggle("is-active", isActive);
       button.setAttribute("aria-pressed", String(isActive));
     });
 
     renderNotes();
   }
 
   if (noteTypeField && checklistBuilder) {
     noteTypeField.addEventListener("change", updateChecklistVisibility);
     updateChecklistVisibility();
   }
 
   if (addChecklistItemButton && checklistItems) {
     addChecklistItemButton.addEventListener("click", function () {
       checklistItems.appendChild(createChecklistRow());
     });
   }
 
   viewButtons.forEach(function (button) {
     button.addEventListener("click", function () {
       setActiveView(button.dataset.view || "active");
     });
   });
 
   if (searchInput) {
     searchInput.addEventListener("input", renderNotes);
   }
 
   if (noteForm) {
     noteForm.addEventListener("submit", function (event) {
       event.preventDefault();
       const noteData = getFormData();
       upsertNote(noteData);
       const didSave = saveNotes();
 
       if (!didSave) {
         notesStatus.textContent = "Note updated in memory, but saving to local storage was not available.";
       } else if (!state.editingNoteId) {
         notesStatus.textContent = "Note saved to this browser.";
       } else {
         notesStatus.textContent = "Note updated and saved to this browser.";
       }
       renderNotes();
       resetForm();
     });
 
     noteForm.addEventListener("reset", function () {
       window.setTimeout(function () {
         resetChecklistBuilder();
         updateChecklistVisibility();
         state.editingNoteId = null;
         updateComposerMode();
         notesStatus.textContent =
           "Form cleared. Create a new note or choose an existing note to edit.";
       }, 0);
     });
   }
 
   if (checklistItems) {
     const firstTextInput = checklistItems.querySelector('input[type="text"]');
     const firstCheckbox = checklistItems.querySelector('input[type="checkbox"]');
 
     if (firstCheckbox) {
       firstCheckbox.disabled = false;
     }
 
     if (firstTextInput) {
       firstTextInput.value = "";
     }
   }
 
   updateComposerMode();
   renderNotes();
   setActiveView("active");
 })();
