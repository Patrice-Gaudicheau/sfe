+const STORAGE_KEY = "notekeeper.notes";
+
+const state = {
+  notes: [],
+  currentView: "active",
+  editingNoteId: null,
+};
+
+const elements = {
+  form: document.getElementById("note-form"),
+  noteId: document.getElementById("note-id"),
+  title: document.getElementById("note-title"),
+  body: document.getElementById("note-body"),
+  notesList: document.getElementById("notes-list"),
+  emptyState: document.getElementById("empty-state"),
+  notesSummary: document.getElementById("notes-summary"),
+  statusMessage: document.getElementById("status-message"),
+  cancelEditButton: document.getElementById("cancel-edit-button"),
+  viewButtons: Array.from(document.querySelectorAll(".view-button")),
+};
+
+function loadNotes() {
+  const stored = localStorage.getItem(STORAGE_KEY);
+
+  if (!stored) {
+    state.notes = [];
+    return;
+  }
+
+  try {
+    const parsed = JSON.parse(stored);
+    state.notes = Array.isArray(parsed)
+      ? parsed
+          .filter((note) => note && typeof note === "object")
+          .map((note) => ({
+            id: typeof note.id === "string" ? note.id : createId(),
+            title: typeof note.title === "string" ? note.title : "",
+            body: typeof note.body === "string" ? note.body : "",
+            archived: Boolean(note.archived),
+            createdAt: typeof note.createdAt === "string" ? note.createdAt : new Date().toISOString(),
+            updatedAt: typeof note.updatedAt === "string" ? note.updatedAt : new Date().toISOString(),
+          }))
+      : [];
+  } catch (error) {
+    state.notes = [];
+    localStorage.removeItem(STORAGE_KEY);
+    setStatus("Stored notes were unreadable, so NoteKeeper started with an empty list.");
+  }
+}
+
+function saveNotes() {
+  localStorage.setItem(STORAGE_KEY, JSON.stringify(state.notes));
+}
+
+function createId() {
+  return `note-${Date.now()}-${Math.random().toString(16).slice(2, 10)}`;
+}
+
+function setStatus(message) {
+  elements.statusMessage.textContent = message;
+}
+
+function clearStatus() {
+  setStatus("");
+}
+
+function resetForm() {
+  state.editingNoteId = null;
+  elements.noteId.value = "";
+  elements.form.reset();
+  elements.cancelEditButton.hidden = true;
+}
+
+function startEdit(noteId) {
+  const note = state.notes.find((item) => item.id === noteId);
+
+  if (!note) {
+    return;
+  }
+
+  state.editingNoteId = note.id;
+  elements.noteId.value = note.id;
+  elements.title.value = note.title;
+  elements.body.value = note.body;
+  elements.cancelEditButton.hidden = false;
+  elements.title.focus();
+  setStatus("Editing note.");
+}
+
+function deleteNote(noteId) {
+  state.notes = state.notes.filter((note) => note.id !== noteId);
+
+  if (state.editingNoteId === noteId) {
+    resetForm();
+  }
+
+  saveNotes();
+  render();
+  setStatus("Note deleted.");
+}
+
+function toggleArchive(noteId) {
+  const note = state.notes.find((item) => item.id === noteId);
+
+  if (!note) {
+    return;
+  }
+
+  note.archived = !note.archived;
+  note.updatedAt = new Date().toISOString();
+
+  if (state.editingNoteId === noteId) {
+    resetForm();
+  }
+
+  saveNotes();
+  render();
+  setStatus(note.archived ? "Note archived." : "Note restored to active notes.");
+}
+
+function getVisibleNotes() {
+  return state.notes
+    .filter((note) => (state.currentView === "archived" ? note.archived : !note.archived))
+    .sort((first, second) => new Date(second.updatedAt) - new Date(first.updatedAt));
+}
+
+function formatDate(value) {
+  const date = new Date(value);
+
+  if (Number.isNaN(date.getTime())) {
+    return "Unknown date";
+  }
+
+  return date.toLocaleString([], {
+    year: "numeric",
+    month: "short",
+    day: "numeric",
+    hour: "numeric",
+    minute: "2-digit",
+  });
+}
+
+function render() {
+  const visibleNotes = getVisibleNotes();
+  elements.notesList.innerHTML = "";
+
+  elements.notesSummary.textContent =
+    state.currentView === "archived"
+      ? `${visibleNotes.length} archived ${visibleNotes.length === 1 ? "note" : "notes"}`
+      : `${visibleNotes.length} active ${visibleNotes.length === 1 ? "note" : "notes"}`;
+
+  if (visibleNotes.length === 0) {
+    elements.emptyState.hidden = false;
+    elements.emptyState.textContent =
+      state.currentView === "archived"
+        ? "No archived notes yet. Archive notes to store them out of the main view."
+        : "No active notes yet. Create a note to get started.";
+    return;
+  }
+
+  elements.emptyState.hidden = true;
+
+  visibleNotes.forEach((note) => {
+    const article = document.createElement("article");
+    article.className = "note-card";
+
+    const title = note.title.trim() || "Untitled note";
+    const bodyContent = note.body.trim();
+    const bodyMarkup = bodyContent
+      ? `<p>${escapeHtml(bodyContent)}</p>`
+      : '<p class="body-placeholder">No note text.</p>';
+
+    article.innerHTML = `
+      <div class="note-card-header">
+        <div>
+          <h3>${escapeHtml(title)}</h3>
+        </div>
+        <div class="note-card-date">Updated ${escapeHtml(formatDate(note.updatedAt))}</div>
+      </div>
+      ${bodyMarkup}
+      <div class="note-card-actions">
+        <button type="button" data-action="edit" data-id="${note.id}">Edit</button>
+        <button type="button" class="archive-button" data-action="archive" data-id="${note.id}">
+          ${note.archived ? "Restore" : "Archive"}
+        </button>
+        <button type="button" class="delete-button" data-action="delete" data-id="${note.id}">
+          Delete
+        </button>
+      </div>
+      <div class="card-footer">Created ${escapeHtml(formatDate(note.createdAt))}</div>
+    `;
+
+    elements.notesList.appendChild(article);
+  });
+}
+
+function escapeHtml(value) {
+  const replacements = {
+    "&": "&amp;",
+    "<": "&lt;",
+    ">": "&gt;",
+    '"': "&quot;",
+    "'": "&#39;",
+  };
+
+  return value.replace(/[&<>"']/g, (character) => replacements[character]);
+}
+
+function handleSubmit(event) {
+  event.preventDefault();
+
+  const title = elements.title.value.trim();
+  const body = elements.body.value.trim();
+  const timestamp = new Date().toISOString();
+
+  if (!title && !body) {
+    setStatus("Enter a title, note text, or both before saving.");
+    elements.body.focus();
+    return;
+  }
+
+  if (state.editingNoteId) {
+    const note = state.notes.find((item) => item.id === state.editingNoteId);
+
+    if (!note) {
+      resetForm();
+      render();
+      setStatus("The note being edited was not found.");
+      return;
+    }
+
+    note.title = title;
+    note.body = body;
+    note.updatedAt = timestamp;
+    setStatus("Note updated.");
+  } else {
+    state.notes.unshift({
+      id: createId(),
+      title,
+      body,
+      archived: false,
+      createdAt: timestamp,
+      updatedAt: timestamp,
+    });
+    setStatus("Note created.");
+  }
+
+  saveNotes();
+  resetForm();
+  render();
+}
+
+function handleListClick(event) {
+  const button = event.target.closest("button[data-action]");
+
+  if (!button) {
+    return;
+  }
+
+  const { action, id } = button.dataset;
+
+  if (action === "edit") {
+    startEdit(id);
+    return;
+  }
+
+  if (action === "archive") {
+    toggleArchive(id);
+    return;
+  }
+
+  if (action === "delete") {
+    deleteNote(id);
+  }
+}
+
+function handleViewChange(event) {
+  const button = event.currentTarget;
+  const nextView = button.dataset.view;
+
+  if (!nextView || nextView === state.currentView) {
+    return;
+  }
+
+  state.currentView = nextView;
+
+  elements.viewButtons.forEach((viewButton) => {
+    const isActive = viewButton.dataset.view === nextView;
+    viewButton.classList.toggle("is-active", isActive);
+    viewButton.setAttribute("aria-pressed", String(isActive));
+  });
+
+  clearStatus();
+  render();
+}
+
+function bindEvents() {
+  elements.form.addEventListener("submit", handleSubmit);
+  elements.notesList.addEventListener("click", handleListClick);
+  elements.cancelEditButton.addEventListener("click", () => {
+    resetForm();
+    setStatus("Edit canceled.");
+  });
+  elements.viewButtons.forEach((button) => {
+    button.addEventListener("click", handleViewChange);
+  });
+}
+
+function init() {
+  loadNotes();
+  bindEvents();
+  render();
+}
+
+init();

+# NoteKeeper
+
+NoteKeeper is a small browser-only notes app that runs directly from the `app/` folder with no build step, server, or dependencies.
+
+## Run the app
+
+1. Open `index.html` in a modern browser.
+2. Start creating notes immediately.
+
+## Current features
+
+- Create plain text notes with an optional title and body
+- Edit existing notes
+- Delete notes
+- Archive and restore notes
+- Persist notes in `localStorage` using the `notekeeper.notes` key
+- Recover safely if stored note data is missing or malformed
+- Switch between active notes and archived notes
+
+## Manual verification
+
+1. Open `index.html`.
+2. Create a note with a title and body, then save it.
+3. Create another note with only a title or only body text and confirm it still renders correctly.
+4. Edit an existing note and save the changes.
+5. Archive a note and switch to the archived view to confirm it appears there.
+6. Restore the archived note back to active notes.
+7. Delete a note and confirm it is removed.
+8. Reload the page and confirm the remaining notes are still present.
+9. Optionally, inspect browser storage and confirm NoteKeeper uses the `notekeeper.notes` localStorage entry.
